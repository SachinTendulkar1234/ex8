// server.js const express = require('express'); const mongoose = require('mongoose'); const cors = require('cors'); require('dotenv').config(); const app = express(); const PORT = process.env.PORT || 3000; app.use(cors()); app.use(express.json()); // MongoDB connection mongoose.connect(process.env.MONGO_URI) .then(() => console.log("✅ MongoDB Connected")) .catch(err => console.error("❌ DB Error:", err)); // Expense Schema const expenseSchema = new mongoose.Schema({ category: String, description: String, amount: Number, date: String }); const Expense = mongoose.model('Expense', expenseSchema); // ➕ Add Expense app.post('/expenses', async (req, res) => { try { const expense = await Expense.create(req.body); res.status(201).json(expense); } catch (err) { res.status(400).json({ error: err.message }); } }); // 📜 Get All Expenses app.get('/expenses', async (req, res) => { const expenses = await Expense.find().sort({ date: -1 }); res.json(expenses); }); // ❌ Delete Expense app.delete('/expenses/:id', async (req, res) => { const deleted = await Expense.findByIdAndDelete(req.params.id); deleted ? res.json({ message: "Expense Deleted" }) : res.status(404).json({ message: "Not Found" }); }); app.listen(PORT, () => console.log(`🚀 Server running on port ${PORT}`)); 💰 Expense Tracker
